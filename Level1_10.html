import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats

# 標準正規分布のパラメータ
mean = 0  # 平均
std_dev = 1  # 標準偏差

# xの範囲
x = np.linspace(-4, 4, 1000)

# 標準正規分布の確率密度関数
y = stats.norm.pdf(x, mean, std_dev)

# LCLとUCLの計算（3シグマ）
LCL = mean - 3 * std_dev
UCL = mean + 3 * std_dev

# グラフの作成
plt.figure(figsize=(10, 5))
plt.plot(x, y, label='distribución normal estándar', color='blue')

# LCLとUCLの塗り分け
plt.fill_between(x, y, where=(x <= LCL), color='red', alpha=0.5, label='LCL')
plt.fill_between(x, y, where=(x >= UCL), color='red', alpha=0.5)

# LCLとUCLのラインを追加
plt.axvline(LCL, color='red', linestyle='--', label='LCL (3σ)')
plt.axvline(UCL, color='red', linestyle='--', label='UCL (3σ)')

# グリッド、x軸、y軸、タイトルを消す
plt.grid(False)
plt.xticks([])  # x軸の目盛りを消す
plt.yticks([])  # y軸の目盛りを消す
plt.title('')  # タイトルを消す

# 凡例を表示
plt.legend()

# グラフを表示
plt.show()
