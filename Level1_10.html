import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats

# 
mean = 0  # 
std_dev = 1  # 

# x
x = np.linspace(-4, 4, 1000)

# 
y = stats.norm.pdf(x, mean, std_dev)

# LCLUCL3
LCL = mean - 3 * std_dev
UCL = mean + 3 * std_dev

# 
plt.figure(figsize=(10, 5))
plt.plot(x, y, label='distribucin normal estndar', color='blue')

# LCLUCL
plt.fill_between(x, y, where=(x <= LCL), color='red', alpha=0.5, label='LCL')
plt.fill_between(x, y, where=(x >= UCL), color='red', alpha=0.5)

# LCLUCL
plt.axvline(LCL, color='red', linestyle='--', label='LCL (3)')
plt.axvline(UCL, color='red', linestyle='--', label='UCL (3)')

# xy
plt.grid(False)
plt.xticks([])  # x
plt.yticks([])  # y
plt.title('')  # 

#
