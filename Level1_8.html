from fractions import Fraction

#
def uniform_distribution_properties(a, b):
    # E(X) = (a + b) / 2
    mean = Fraction(a + b, 2)
    # E(X^2) = (a^2 + ab + b^2) / 3
    mean_square = Fraction(a**2 + a*b + b**2, 3)
    # V(X) = ((b - a)^2) / 12
    variance = Fraction((b - a) ** 2, 12)
    return mean, mean_square, variance

# X ~ U(0, 4)
a_X, b_X = 0, 4
E_X, E_X2, V_X = uniform_distribution_properties(a_X, b_X)

# Y ~ U(0, 6)
a_Y, b_Y = 0, 6
E_Y, E_Y2, V_Y = uniform_distribution_properties(a_Y, b_Y)

# X + Y
E_X_plus_Y = E_X + E_Y
V_X_plus_Y = V_X + V_Y

# X - Y
E_X_minus_Y = E_X - E_Y
V_X_minus_Y = V_X + V_Y

2種類の部品A(寸法X)、B(寸法Y)を別々に製造し、これらを組み合わせて製品(寸法Z=X+Y)を作る工程がある。

部品AのXは正規分布 $ N(30, 2^2) $ に従い、部品BのYは正規分布 $ N(10, 1^2) $ に従う。

組み合わせ作業では、Xの大きな部品AにはYの小さな部品Bを組み合わせ、Xの小さな部品AにはYの大きな部品Bを組み合わせており、その組み合わせの実施状況はXとYの母相関係数で $ -0.75 $であることがわかっている。

寸法Zの分散を求めよ。

Hay un proceso de fabricación en el que se producen por separado dos tipos de componentes. A (dimensión X) y B (dimensión Y), y luego se combinan para formar un producto como la dimensión $ Z = X + Y $.

La dimensión X del componente A sigue una distribución normal $ N(30,2^2) $, y la dimension Y del componente B sigue una distribución normal $ N(10, 1^2) $.

Durante el proceso de ensamblaje, los componentes A con un valor alto de X se combinan con los componponetes B con un valor bajo de Y, y viceversa. Se sabe que el coeficiente de correlación poblacional entre X e Y es de $ -0.75 $.

Determinar la varianza de la dimensión Z.

$ \rho(X,Y) = \frac{Cov(X,Y)}{\sqrt{V(X) \cdot V(Y)}}$

# Datos de entrada
sigma_X_squared = 4  # Varianza de X (σ_X^2)

sigma_Y = math.sqrt(sigma_Y_squared)  # Desviación estándar de Y

# Cálculo de la varianza de Z

# Función para calcular la varianza del volumen del espacio vacio
def probability_defective(sigma_Z_squared):
    if sigma_Z_squared <= 0:
        return 1.0  # Valores de varianza menores o iguales a 0 son inválidos

    # Desviación estándar del volumen de espacio vacío Z
    sigma_Z = np.sqrt(sigma_Z_squared)

    # Distribución de probabilidad del volumen del espacio vacío Z
    prob_within_range = stats.norm.cdf(upper_bound, mu_Z_new, sigma_Z) - stats.norm.cdf(lower_bound, mu_Z_new, sigma_Z)

    # Probabilidad de productos defectuosos
    prob_defective = 1 - prob_within_range

    return prob_defective - (1 - required_probability)

# Para garantizar que la varianza sea mayor que 0, se establece un valor mínimo
lower_bound_variance = 0.01

# Evaluar la función para valores mayores o iguales a 0 y buscar la solución
optimal_variance = brentq(probability_defective, lower_bound_variance, 1000)

# Resultados
optimal_std_dev = np.sqrt(optimal_variance)
optimal_prob_defective = probability_defective(optimal_variance) + (1 - required_probability)

# Mostrar resultados
print(f"Varianza poblacional del volumen del espacio vacío: {optimal_variance:.1f} ml^2")
