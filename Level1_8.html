from fractions import Fraction

def uniform_distribution_properties(a, b):
    # E(X) = (a + b) / 2
    mean = Fraction(a + b, 2)
    # E(X^2) = (a^2 + ab + b^2) / 3
    mean_square = Fraction(a**2 + a*b + b**2, 3)
    # V(X) = ((b - a)^2) / 12
    variance = Fraction((b - a) ** 2, 12)
    return mean, mean_square, variance

# X ~ U(0, 4)

import scipy.stats as stats

# Configurar los grados de libertad
phi = 10  # Grados de libertad de la distribucin t
phi1 = 1  # Grados de libertad 1 de la distribucin F
phi2 = phi  # Grados de libertad 2 de la distribucin F

# Generar muestras de la distribucin t
t_samples = np.random.standard_t(df=phi, size=100000)
t_squared_samples = t_samples**2  # Elevar al cuadrado los valores de t

# Calcular la funcin de densidad de probabilidad terica de la distribucin F
x = np.linspace(0, 5, 1000)
f_pdf = stats.f.pdf(x, phi1, phi2)

# Dibujar el histograma
plt.figure(figsize=(8, 5))
plt.hist(t_squared_samples, bins=500, density=True, alpha=0.6, label=r'$t^2$ samples')
plt.plot(x, f_pdf, 'r-', label=r'$F(1, \phi)$ distribution')
plt.xlim(0,5)

# Decorar el grfico
plt.xlabel('Value')
plt.ylabel('Density')
plt.title(r'Visualization of $t^2 \sim F(1, \phi)$')
plt.legend()
plt.grid()
plt.show()

import scipy.stats as stats

# Configuracin de parmetros
lambda_param = 1 / 100  # Tasa de fallos 
t1 = 100  # Lmite inferior de la vida til
t2 = 200  # Lmite superior de la vida til

# Calcular la probabilidad utilizando la funcin acumulada (CDF)
# de la distribucin exponencial
# CDF(x) = 1 - e^(-x)
prob_t1 = 1 - np.exp(-lambda_param * t1)
prob_t2 = 1 - np.exp(-lambda_param * t2)

# Probabilidad de que la vida til est entre 100 y 200 horas
prob = prob_t2 - prob_t1

# Mostrar el resultado
print(f"la probabilidad de que la vida til est entre 100 y 200 horas: {prob:.2f}")
