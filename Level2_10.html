def calculate_expected_value_and_variance(prizes, probabilities):
    # 期待値を計算
    expected_value = sum(p * x for p, x in zip(probabilities, prizes))

    # 分散を計算
    variance = sum(p * (x - expected_value) ** 2 for p, x in zip(probabilities, prizes))

    return expected_value, variance

# くじAのデータ
prizes_A = [100, 10, 5, 0]  # 賞金（0は外れを意味する）
probabilities_A = [0.001, 0.020, 0.040, 0.939]  # 確率（外れの確率は1から全ての当たりの確率を引いたもの）

# くじBのデータ
prizes_B = [20, 5, 1, 0]  # 賞金（0は外れを意味する）
probabilities_B = [0.01, 0.04, 0.10, 0.85]  # 確率（外れの確率は1から全ての当たりの確率を引いたもの）

# くじAの期待値と分散を計算

print(f"くじAの期待値: {expected_value_A}")
print(f"くじAの分散: {variance_A}")

# くじBの期待値と分散を計算
