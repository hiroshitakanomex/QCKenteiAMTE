import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import binom

# パラメータ設定
n = 20  # サンプル数
c = 3   # 合格判定個数

# 不適合品率 p の範囲
p_values = np.linspace(0, 0.5, 101)  # 0から1までの範囲を100分割

# 合格確率 L の計算
L_values = []

for p in p_values:
    # 不適合品数 k が c より少ない場合の合格確率を計算
    L = binom.cdf(c, n, p)
    L_values.append(L)

# グラフの描画
plt.figure(figsize=(10, 6))
plt.plot(p_values, L_values, linestyle='-', color='b')
plt.title('OC Curve (n={}, c={})'.format(n, c))
plt.xlabel('La tasa de productos defectuosas del lote p')
plt.ylabel('La probabilidad de aceptación del lote L')
plt.grid(True)
plt.ylim(0, 1)
plt.xlim(0, 0.5)
plt.axhline(y=0.867, xmin=0, xmax=0.2, color='r', linestyle='--', label='Consumer Risk (β=0.867)')
plt.axvline(x=0.1, ymin=0, ymax=0.867, color='g', linestyle='--', label='Producer Risk (α=0.1)')
plt.axhline(y=0.411, xmin=0, xmax=0.4, color='r', linestyle='--', label='Consumer Risk (β=0.411)')

import math

def calculate_sample_size(alpha, beta, sigma, upper_limit, lower_limit):
    z_alpha = stats.norm.ppf(alpha)  # 生産者リスクαに対応するZ値
    z_beta = stats.norm.ppf(beta)        # 消費者リスクβに対応するZ値

    print(z_alpha, z_beta)

    # 平均値がmuの正規分布から得られるサンプルの標準誤差
    se = sigma / math.sqrt(lower_limit)

    # 合格判定値（上限）に対するZ値
    z_upper_limit = (upper_limit - mu) / se

    # サンプルの大きさnの計算
    n = ((z_alpha + z_beta) / (z_upper_limit - z_beta)) ** 2
    n = math.ceil(n)  # nは整数である必要があるため切り上げ

    # 下限合格判定値を求める
    c_lower = (upper_limit*z_beta + lower_limit*z_alpha)/(z_beta+z_alpha)


    return n, c_lower

import numpy as np
import scipy.stats as stats
import matplotlib.pyplot as plt

# パラメータ設定
alpha = 0.05    # 生産者のリスク
beta = 0.10     # 消費者のリスク
sigma = 2.4     # 標準偏差（kg）
upper_limit = 200.0  # 平均値がこの値以上で合格
lower_limit = 197.0  # 平均値がこの値以下で不合格

# 正常なプロセスの平均
mu_normal = 200.0

# 不良なプロセスの平均
mu_defective = 197.0

# サンプルの大きさnと下限合格判定値cを求める関数
def find_sample_size_and_acceptance_limit(alpha, beta, mu_normal, mu_defective, sigma):
    n = 1
    while True:
        z_alpha = stats.norm.ppf(1 - alpha)
        z_beta = stats.norm.ppf(beta)

        # 下限合格判定値 c の計算
        c = mu_normal - (z_alpha * sigma / np.sqrt(n))

        # OC曲線を描くためのパラメータ
        p_accept = stats.norm.cdf((c - mu_defective) / (sigma / np.sqrt(n)))

        if p_accept >= (1 - beta):
            break
        n += 1

    return n, c

# サンプルの大きさnと下限合格判定値cを求める
n, c = find_sample_size_and_acceptance_limit(alpha, beta, mu_normal, mu_defective, sigma)

# グラフ作成
sample_sizes = np.arange(1, n + 10)
acceptance_probs = [stats.norm.cdf((c - mu_defective) / (sigma / np.sqrt(n))) for n in sample_sizes]

plt.plot(sample_sizes, acceptance_probs, label='Acceptance Probability')
plt.axhline(1 - beta, color='r', linestyle='--', label='Consumer\'s Risk Threshold (1 - beta)')
plt.xlabel('Sample Size (n)')
plt.ylabel('Acceptance Probability')
plt.title('Operating Characteristic (OC) Curve')
plt.legend()
plt.grid(True)
plt.show()

# 結果の表示
print(f"サンプルの大きさ (n): {n}")
print(f"下限合格判定値 (c): {c:.2f}")
