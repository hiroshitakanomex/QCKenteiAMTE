    se = sigma / math.sqrt(lower_limit)

    # n

import numpy as np
import scipy.stats as stats
import matplotlib.pyplot as plt

# 
alpha = 0.05    # 
beta = 0.10     # 
sigma = 2.4     # kg
upper_limit = 200.0  # 
lower_limit = 197.0  # 

# 
mu_normal = 200.0

# 
mu_defective = 197.0

# nc
def find_sample_size_and_acceptance_limit(alpha, beta, mu_normal, mu_defective, sigma):
    n = 1
    while True:
        z_alpha = stats.norm.ppf(1 - alpha)
        z_beta = stats.norm.ppf(beta)

        #  c 
        c = mu_normal - (z_alpha * sigma / np.sqrt(n))

        # OC
        p_accept = stats.norm.cdf((c - mu_defective) / (sigma / np.sqrt(n)))

        if p_accept >= (1 - beta):
            break
        n += 1

    return n, c

# nc
n, c = find_sample_size_and_acceptance_limit(alpha, beta, mu_normal, mu_defective, sigma)

# 
sample_sizes = np.arange(1, n + 10)
acceptance_probs = [stats.norm.cdf((c - mu_defective) / (sigma / np.sqrt(n))) for n in sample_sizes]

plt.plot(sample_sizes, acceptance_probs, label='Acceptance Probability')
plt.axhline(1 - beta, color='r', linestyle='--', label='Consumer\'s Risk Threshold (1 - beta)')
plt.xlabel('Sample Size (n)')
plt.ylabel('Acceptance Probability')
plt.title('Operating Characteristic (OC) Curve')
plt.legend()
plt.grid(True)
plt.show()

# 
print(f" (n): {n}")
print(f" (c): {c:.2f}")
