import numpy as np
import pandas as pd
from scipy import stats

# データを設定します
# 各水準ごとのデータ
data = {
    'Nivel': [1]*4 + [2]*4 + [3]*4 + [4]*4,
    'Valor': [2,4,3,5,6,5,3,4,8,6,7,5,3,4,5,2]
}

df = pd.DataFrame(data)

# 総平均を計算
total_mean = df['Valor'].mean()

# 各水準の平均を計算
nivel_averages = df.groupby('Nivel')['Valor'].mean()

# 総平方和 (SST) を計算
SST = np.sum((df['Valor'] - total_mean)**2)

# 因子Aの平方和 (SSA) を計算
SSA = np.sum(df.groupby('Nivel')['Valor'].count() * (nivel_averages - total_mean)**2)

# 誤差平方和 (SSE) を計算
SSE = SST - SSA

# 自由度を計算
n = len(df['Valor'])  # サンプルサイズ
k = df['Nivel'].nunique()  # 因子の水準数

df_factor = k - 1  # 因子の自由度
df_error = n - k  # 誤差の自由度

# 分散を計算
MSA = SSA / df_factor  # 因子Aの平均平方
MSE = SSE / df_error  # 誤差の平均平方

# 分散比 (F値) を計算
F_value = MSA / MSE

# F検定のp値を計算
p_value = 1 - stats.f.cdf(F_value, df_factor, df_error)

# 有意水準5%でのF検定の規格限界値を計算
alpha = 0.05
F_critical = stats.f.ppf(1 - alpha, df_factor, df_error)

# 結果を表示
print(f"総平方和 (SST): {SST:.2f}")
print(f"因子Aの平方和 (SSA): {SSA:.2f}")
print(f"誤差平方和 (SSE): {SSE:.2f}")
print(f"因子Aの自由度: {df_factor}")
print(f"誤差の自由度: {df_error}")
print(f"因子Aの分散比 (F値): {F_value:.2f}")
print(f"F検定のp値: {p_value:.4f}")
print(f"有意水準5%でのF検定の規格限界値: {F_critical:.2f}")

# 有意水準5%でのF検定の結果
alpha = 0.05
if p_value < alpha:
    print(f"因子Aの効果は有意です（p < {alpha}）")
else:
    print(f"因子Aの効果は有意ではありません（p >= {alpha}）")

import pandas as pd
import statsmodels.api as sm
from statsmodels.formula.api import ols
import matplotlib.pyplot as plt

# データの定義
data = {
    'A': ['A1', 'A1', 'A1', 'A1', 'A2', 'A2', 'A2', 'A2'],
    'B': ['B1', 'B2', 'B3', 'B4', 'B1', 'B2', 'B3', 'B4'],
    'Value': [6, 10, 8, 4, 8, 12, 12, 8]
}
df = pd.DataFrame(data)

# データの検証
if df.isnull().values.any():
    print("データに欠損値が含まれています。")
else:
    print("データに欠損値はありません。")

# データの構造式を定義（交互作用項を除く）
model = ols('Value ~ C(A) + C(B)', data=df).fit()
anova_table = sm.stats.anova_lm(model, typ=2)

# 総平均を計算
total_mean = df['Value'].mean()

# 因子Aの第2水準における効果の推定
mean_A2 = df[df['A'] == 'A2']['Value'].mean()
mean_A1 = df[df['A'] == 'A1']['Value'].mean()
effect_A2 = mean_A2 - total_mean

# 因子Aの第1水準、第2水準における効果の期待値の平均平方とその期待値
mean_square_A = anova_table['sum_sq']['C(A)'] / anova_table['df']['C(A)']
mean_square_B = anova_table['sum_sq']['C(B)'] / anova_table['df']['C(B)']
mean_square_error = anova_table['sum_sq']['Residual'] / anova_table['df']['Residual']

# 結果の表示
print("ANOVA Table:")
print(anova_table)
print("\nEffect of A at Level 2:", effect_A2)
print("\nMean Square of Factor A:", mean_square_A)
print("Mean Square of Factor B:", mean_square_B)
print("Mean Square of Error:", mean_square_error)

fig, ax = plt.subplots()

import numpy as np
import pandas as pd
from scipy import stats

# データを設定します
# 因子Aは2水準、因子Bは4水準、繰り返しなし
data = {
    'A': ['A1', 'A1', 'A1', 'A1', 'A2', 'A2', 'A2', 'A2'],
    'B': ['B1', 'B2', 'B3', 'B4', 'B1', 'B2', 'B3', 'B4'],
    'Value': [6,10,8,4,8,12,12,8]  # データ
}

df = pd.DataFrame(data)

# 総平均を計算
total_mean = df['Value'].mean()

print(total_mean)

# 因子Aの平均を計算
A_means = df.groupby('A')['Value'].mean()

# 因子Bの平均を計算
B_means = df.groupby('B')['Value'].mean()

# 総平方和 (SST) を計算
SST = np.sum((df['Value'] - total_mean)**2)

# 因子Aの平方和 (SSA) を計算
SSA = np.sum(df.groupby('A').count()['Value'] * (A_means - total_mean)**2)

# 因子Bの平方和 (SSB) を計算
SSB = np.sum(df.groupby('B').count()['Value'] * (B_means - total_mean)**2)

# 誤差平方和 (SSE) を計算
SSE = SST - SSA - SSB

# 自由度を計算
n = len(df['Value'])  # 総サンプルサイズ
a = df['A'].nunique()  # 因子Aの水準数
b = df['B'].nunique()  # 因子Bの水準数

df_A = a - 1  # 因子Aの自由度
df_B = b - 1  # 因子Bの自由度
df_error = (a * b) - (a + b - 1)  # 誤差の自由度

# 分散を計算
MSA = SSA / df_A  # 因子Aの平均平方
MSB = SSB / df_B  # 因子Bの平均平方
MSE = SSE / df_error  # 誤差の平均平方

# 分散比 (F値) を計算
F_A = MSA / MSE
F_B = MSB / MSE

# F検定のp値を計算
p_value_A = 1 - stats.f.cdf(F_A, df_A, df_error)
p_value_B = 1 - stats.f.cdf(F_B, df_B, df_error)

# 結果を表示
print(f"総平方和 (SST): {SST:.2f}")
print(f"因子Aの平方和 (SSA): {SSA:.2f}")
print(f"因子Bの平方和 (SSB): {SSB:.2f}")
print(f"誤差平方和 (SSE): {SSE:.2f}")
print(f"因子Aの自由度: {df_A}")
print(f"因子Bの自由度: {df_B}")
print(f"誤差の自由度: {df_error}")
print(f"因子Aの分散比 (F値): {F_A:.2f}")
print(f"因子Bの分散比 (F値): {F_B:.2f}")
print(f"因子AのF検定のp値: {p_value_A:.4f}")
print(f"因子BのF検定のp値: {p_value_B:.4f}")

# 有意水準5%でのF検定の結果
alpha = 0.05
F_critical_A = stats.f.ppf(1 - alpha, df_A, df_error)
F_critical_B = stats.f.ppf(1 - alpha, df_B, df_error)

print(f"因子Aの有意水準5%でのF検定の規格限界値: {F_critical_A:.2f}")
print(f"因子Bの有意水準5%でのF検定の規格限界値: {F_critical_B:.2f}")

if F_A > F_critical_A:
    print(f"因子Aの効果は有意です（F値 > 規格限界値: {F_A:.2f} > {F_critical_A:.2f}）")
else:
    print(f"因子Aの効果は有意ではありません（F値 <= 規格限界値: {F_A:.2f} <= {F_critical_A:.2f}）")

if F_B > F_critical_B:
