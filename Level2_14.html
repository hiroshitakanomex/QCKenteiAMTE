import numpy as np
import pandas as pd
from scipy import stats

# 
# 
data = {
    'Nivel': [1]*4 + [2]*4 + [3]*4 + [4]*4,
    'Valor': [2,4,3,5,6,5,3,4,8,6,7,5,3,4,5,2]
}

df = pd.DataFrame(data)

# 
total_mean = df['Valor'].mean()

# 
nivel_averages = df.groupby('Nivel')['Valor'].mean()

#  (SST) 
SST = np.sum((df['Valor'] - total_mean)**2)

# A (SSA) 
SSA = np.sum(df.groupby('Nivel')['Valor'].count() * (nivel_averages - total_mean)**2)

#  (SSE) 
SSE = SST - SSA

# 
n = len(df['Valor'])  # 
k = df['Nivel'].nunique()  # 

df_factor = k - 1  # 
df_error = n - k  # 

# 
MSA = SSA / df_factor  # A
MSE = SSE / df_error  # 

#  (F) 
F_value = MSA / MSE

# Fp
p_value = 1 - stats.f.cdf(F_value, df_factor, df_error)

# 5%F
alpha = 0.05
F_critical = stats.f.ppf(1 - alpha, df_factor, df_error)

# 
print(f" (SST): {SST:.2f}")
print(f"A (SSA): {SSA:.2f}")
print(f" (SSE): {SSE:.2f}")
print(f"A: {df_factor}")
print(f": {df_error}")
print(f"A (F): {F_value:.2f}")
print(f"Fp: {p_value:.4f}")
print(f"5%F: {F_critical:.2f}")

# 5%F
alpha = 0.05
if p_value < alpha:
    print(f"Ap < {alpha}")
else:
    print(f"Ap >= {alpha}")
