import numpy as np
import matplotlib.pyplot as plt

# 
t = np.linspace(0, 100, 500)

# 
early_failure_rate = 0.05 * np.exp(-t / 10)
random_failure_rate = 0.001 * np.ones_like(t)
wear_out_failure_rate = 0.02 * np.exp((t - 70) / 10)

# 
failure_rate = early_failure_rate + random_failure_rate + wear_out_failure_rate

# 
plt.figure(figsize=(10, 6))

# 
plt.axvspan(0, 20, color='red', alpha=0.1, label='Early Failure Period')
# 
plt.axvspan(20, 70, color='blue', alpha=0.1, label='Random Failure Period')
# 
plt.axvspan(70, 100, color='green', alpha=0.1, label='Wear-out Failure Period')

# 
plt.plot(t, early_failure_rate, label='Early Failures', color='red')
plt.plot(t, random_failure_rate, label='Random Failures', color='blue')
plt.plot(t, wear_out_failure_rate, label='Wear-out Failures', color='green')
plt.plot(t, failure_rate, label='Total Failure Rate', linewidth=2, color='black')

# 
plt.title('Bathtub Curve')
plt.xlabel('Time')
plt.ylabel('Failure Rate')
plt.legend()
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt

# 
failures = {
    "Product 1": [1, 4, 8],
    "Product 2": [3],
    "Product 3": [7, 9],
    "Product 4": [9, 12]
}

# 
failure_counts = {k: len(v) for k, v in failures.items()}

# 12
total_operating_time = 4 * 12  # 4 * 12

# 
total_failures = sum(failure_counts.values())

# MTBF
MTBF = total_operating_time / total_failures

# 
repair_times = {
    "Product 1": [0.5, 0.5, 0.5],
    "Product 2": [0.5],
    "Product 3": [0.5, 0.5],
    "Product 4": [0.5, 0.5]
}

# 
total_repair_time = sum([sum(v) for v in repair_times.values()])

# MTTR
MTTR = total_repair_time / total_failures

# 
availability = MTBF / (MTBF + MTTR)

# 
failure_rate = 1 / MTBF

# 
colors = ['skyblue', 'lightgreen', 'lightcoral', 'plum']
repair_color = 'lightgray'  # 

# 
plt.figure(figsize=(12, 6))

for i, (product, failure_times) in enumerate(failures.items()):
    start = 0
    for j, ft in enumerate(failure_times):
        plt.text(start + (ft - start)+0.2, i + 1, f'{ft}', va='center', ha='center', color='black', fontsize=20)
        plt.barh(i + 1, ft - start, left=start, height=0.6, color=colors[i % len(colors)], edgecolor='black')
        start = ft
    # 
    plt.barh(i + 1, 12 - start, left=start, height=0.6, color=repair_color, edgecolor='black')

# 
y_pos = np.arange(len(failures)) + 1
plt.yticks(y_pos, failures.keys())
plt.xlabel('Time (months)')
plt.title('Life of Each Product with Multiple Failures')
plt.axvline(MTBF, color='red', linestyle='--', label=f'MTBF = {MTBF:.2f} months')
plt.legend()

# 
plt.grid(True)
plt.show()

# MTBFMTTR
print(f'Mean Time Between Failures (MTBF): {MTBF:.2f} months')
print(f'Mean Time To Repair (MTTR): {MTTR:.2f} months')
print(f'Availability: {availability:.4f}')
print(f'Failure Rate: {failure_rate:.4f} failures per month')
