import scipy.stats as stats

def calculate_chi_square(df, P):
    # P に対応するχ二乗値を計算する
    chi2_value = stats.chi2.isf(P, df)
    return chi2_value

import scipy.stats as stats


def plot_chi_square(df, P):
    # χ二乗値の範囲を設定
    x = np.linspace(0, stats.chi2.ppf(0.99, df), 1000)

    # χ二乗分布の確率密度関数を計算
    y = stats.chi2.pdf(x, df)

    # グラフを作成
    plt.figure(figsize=(10, 6))
    plt.plot(x, y, 'b-', lw=2, label=f'densidad χ-cuadrado (df={df})')

    # 上側確率Pに対応するχ二乗値を計算
    chi2_value = stats.chi2.isf(P, df)

    # 上側確率Pに対応するχ二乗値の垂直線を描く
    plt.axvline(chi2_value, color='r', linestyle='--', lw=2, label=f'valor χ-quadrado (P={P})')

    # グラフのタイトルとラベルを設定
    plt.title(f'Distribución  χ-cuadrado (grados de libertad = {df})')
    plt.xlabel('χ-cuadroados')
    plt.ylabel('Densidad de probabilidad')
    plt.legend()

    # グラフを表示
    plt.grid()
    plt.show()

# 自由度と上側確率を指定する
df = 5  # 自由度
P = 0.05  # 上側確率 (例: 0.05 = 5%)

# χ二乗分布のグラフを作成する

from scipy.stats import binom

def binomial_probability(P, n, x):
    # P: 母不適合率
    # n: サンプルの大きさ
    # x: 不適合品の数

    # 二項分布の確率質量関数を計算
    px = binom.pmf(x, n, P)
    return px

import scipy.stats as stats

# 充填前の液体製品の母平均と母分散
mu_liquid = 1010.01  # ml
var_liquid = 3.0**2

# 充填後のポリ容器の母平均と母分散
mu_container = 1022.0  # ml
var_container = 4.0**2

# 空隙部の母平均と母分散の計算
mu_empty = mu_container - mu_liquid
var_empty = var_container + var_liquid

# 空隙部が0以下になる確率を求める
# 標準化してから計算する
std_dev_empty = np.sqrt(var_empty)  # 標準偏差の計算
std_dev_empty = std_dev_empty*1.0
# 標準正規分布での確率を計算する
overflow_prob = 1-stats.norm.cdf(0, loc=-mu_empty, scale=std_dev_empty)

# 結果の出力
print(f"ポリ容器から液体があふれる確率: {overflow_prob:.4f}")
