import scipy.stats as stats


def plot_chi_square(df, P):
    # 
    x = np.linspace(0, stats.chi2.ppf(0.99, df), 1000)

    # 
    y = stats.chi2.pdf(x, df)

    # 
    plt.figure(figsize=(10, 6))
    plt.plot(x, y, 'b-', lw=2, label=f'densidad -cuadrado (df={df})')

    # P
    chi2_value = stats.chi2.isf(P, df)

    # P
    plt.axvline(chi2_value, color='r', linestyle='--', lw=2, label=f'valor -quadrado (P={P})')

    # 
    plt.title(f'Distribucin  -cuadrado (grados de libertad = {df})')
    plt.xlabel('-cuadroados')
    plt.ylabel('Densidad de probabilidad')
    plt.legend()

    # 
    plt.grid()
    plt.show()
