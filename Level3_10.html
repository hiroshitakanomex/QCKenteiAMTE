def f (fa, a):
  fa = fa
  A = []

  for d in fa:
    # 
    target_range = d[1]
    mean = d[0]

    while True:
        # 
        std_dev = target_range / 4  # 4

        # 
        sample_data = np.random.normal(mean, std_dev, 5)
        sample_data = np.round(sample_data,1)

        # 
        data_range = np.max(sample_data) - np.min(sample_data)
        data_mean = np.mean(sample_data)
        data_sigma = np.std(sample_data, ddof=1)

        # 
        if abs(data_range - target_range) < 0.01:
          if abs(data_mean - mean) < 0.05:  # 
            break
    A += [sample_data]


  df = pd.DataFrame(A)
  df['xbar'] = df.mean(axis=1)
  df['R'] = df.max(axis=1) - df.min(axis=1)
  df.to_excel('Data3/Kanrizu_' + a + '.xlsx')


  #fig
  fig, ax = plt.subplots(2,figsize=(8,5))
  fig.suptitle(a)

  #subplot

  x = np.arange(1,26)
  y = df['xbar']
  y2 = df['R']
  ym = df['xbar'].mean()
  dm = df['R'].mean()
  ucl = ym+0.577*dm
  lcl = ym-0.577*dm
  UCL_R = 2.114*dm

  ax[0].plot(x,y,marker='o')
  ax[0].axhline(ucl, linestyle='--')
  ax[0].axhline(ym, linestyle='-')
  ax[0].axhline(lcl, linestyle='--')
  ax[0].set_ylim(6.6,8.0)
  ax[0].set_ylabel(r'$\bar{x}$')
  ax[0].text(26.5,ucl,"UCL")
  ax[0].text(26.5,ym,"CL")
  ax[0].text(26.5,lcl,"LCL")


  ax[1].plot(x,y2, marker='x')
  ax[1].axhline(dm, linestyle='-')
  ax[1].axhline(UCL_R,linestyle='--')
  ax[1].set_ylim(0,2.0)
  ax[1].set_ylabel('R')
  ax[1].text(26.5,UCL_R,"UCL")
  ax[1].text(26.5,dm,"CL")

  plt.savefig("Graph3/Kanrizu_" + a + ".png", format="png", dpi=300)

  print(y.mean(), ucl, lcl, data_sigma*3+data_mean)
