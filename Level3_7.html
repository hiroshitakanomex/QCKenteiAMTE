  max = max
  lim_min = lim_min
  lim_max = lim_max

  l = []
  m = len(k)

  for i,j in enumerate(range(30, 30+m)):
    for _ in range(k[i]):
      l += [np.round(random.uniform(j, j+1),1)]


  df71 = pd.DataFrame({'x': l})

  print(len(df71))

  bins = range(30,30+m)

  plt.hist(l, bins=bins, edgecolor="black")
  plt.vlines(lim_min,0,max,colors="black")
  plt.vlines(lim_max,0,max,colors="black")
  plt.xlim(30,30+m)
  plt.ylim(0,max)
  plt.title('Histgram')

  custom_xticks = []
