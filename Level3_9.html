import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats

# 標準正規分布のパラメータ
mean = 0
std_dev = 1

# x軸の範囲を設定
x = np.linspace(-4, 4, 1000)

# 標準正規分布の確率密度関数を計算
y = stats.norm.pdf(x, mean, std_dev)

# Kp = 1.96 に対応する P の範囲
Kp = 1.96
lower_bound = -Kp
upper_bound = Kp

# プロット
plt.plot(x, y, label='Standard Normal Distribution')

# 範囲内を塗りつぶし
x_fill = np.linspace(upper_bound,4, 100)
y_fill = stats.norm.pdf(x_fill, mean, std_dev)
plt.fill_between(x_fill, y_fill, color='lightblue', alpha=0.6, label=f'P (|K| ≤ {Kp})')

# グラフの装飾
plt.title(f'Standard Normal Distribution with Kp={Kp}')
plt.xlabel('x')
plt.ylabel('Probability Density')
plt.axvline(x=upper_bound, color='red', linestyle='--', label=f'Kp = {Kp}')
plt.legend()
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm

# 正規分布のパラメータ
mu = 50
sigma = 2

# 確率の計算
prob_x_greater_52 = 1 - norm.cdf(52, loc=mu, scale=sigma)
prob_x_greater_56 = 1 - norm.cdf(56, loc=mu, scale=sigma)

print(f"P(x > 52): {prob_x_greater_52:.4f}")
print(f"P(x > 56): {prob_x_greater_56:.4f}")

# グラフの作成
x = np.linspace(mu - 4*sigma, mu + 4*sigma, 1000)
y = norm.pdf(x, loc=mu, scale=sigma)

plt.figure(figsize=(10, 6))
plt.plot(x, y, label='Normal Distribution $N(50, 4)$')
plt.fill_between(x, y, where=(x > 52), color='red', alpha=0.5, label='P(x > 52)')
plt.fill_between(x, y, where=(x > 56), color='blue', alpha=0.5, label='P(x > 56)')
plt.title('Normal Distribution and Probability Areas')
plt.xlabel('x')
plt.ylabel('Probability Density')
plt.legend()
plt.grid(True)
plt.show()

P(x > 52): 0.1587
P(x > 56): 0.0013

import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import binom

# パラメータ
p = 0.30  # 不適合品の確率
n = 4     # サンプルサイズ
k = 2     # 不適合品の数

# 二項分布の確率計算
prob_k_2 = binom.pmf(k, n, p)
print(f"P(サンプルの中に不適合品が2個現れる確率): {prob_k_2:.4f}")

# 確率質量関数 (PMF) の計算
k_values = np.arange(0, n + 1)
probabilities = binom.pmf(k_values, n, p)

# グラフの作成
plt.figure(figsize=(10, 6))
plt.bar(k_values, probabilities, color='skyblue', edgecolor='black')
plt.xlabel('不適合品の数')
plt.ylabel('確率')
plt.title('二項分布: 不適合品の数に対する確率')
plt.xticks(k_values)
plt.grid(True)
plt.show()
