import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm

# 
mu = 50
sigma = 2

# 
prob_x_greater_52 = 1 - norm.cdf(52, loc=mu, scale=sigma)
prob_x_greater_56 = 1 - norm.cdf(56, loc=mu, scale=sigma)

print(f"P(x > 52): {prob_x_greater_52:.4f}")
print(f"P(x > 56): {prob_x_greater_56:.4f}")

# 
x = np.linspace(mu - 4*sigma, mu + 4*sigma, 1000)
y = norm.pdf(x, loc=mu, scale=sigma)

plt.figure(figsize=(10, 6))
plt.plot(x, y, label='Normal Distribution $N(50, 4)$')
plt.fill_between(x, y, where=(x > 52), color='red', alpha=0.5, label='P(x > 52)')
plt.fill_between(x, y, where=(x > 56), color='blue', alpha=0.5, label='P(x > 56)')
plt.title('Normal Distribution and Probability Areas')
plt.xlabel('x')
plt.ylabel('Probability Density')
plt.legend()
plt.grid(True)
plt.show()

import numpy as np

from scipy.stats import binom

# 
p = 0.30  # 
n = 4     # 
k = 2     #

probabilities = binom.pmf(k_values, n, p)

#

plt.bar(k_values, probabilities, color='skyblue', edgecolor='black')
plt.xlabel('')
plt.ylabel('')
plt.title(': ')
