def f2 (fa, a):
  fa = fa
  A = []

  k = len(fa)

  for d in fa:
    # 目標のデータ範囲
    target_range = d[1]
    mean = d[0]

    while True:
        # 平均と標準偏差を指定
        std_dev = target_range / 5  # データ範囲の最大値を4倍する

        # サンプルデータを生成
        sample_data = np.random.normal(mean, std_dev, 3)
        sample_data = np.round(sample_data,1)

        # データの範囲を計算
        data_range = np.max(sample_data) - np.min(sample_data)
        data_mean = np.mean(sample_data)
        data_sigma = np.std(sample_data, ddof=1)

        # データ範囲が目標に近い場合はループを抜ける
        if abs(data_range - target_range) < 0.5:
          if abs(data_mean - mean) < 0.5:  # 許容誤差
            break
    A += [sample_data]


  df = pd.DataFrame(A)
  df['xbar'] = df.mean(axis=1)
  df['R'] = df.max(axis=1) - df.min(axis=1)
  df.to_excel('Data4/Kanrizu_' + a + '.xlsx')


  #グラフを表示する領域を，figオブジェクトとして作成。
  fig, ax = plt.subplots(2,figsize=(8,5))
  fig.suptitle(a)

  #グラフを描画するsubplot領域を作成。

  x = np.arange(1,k+1)
  y = df['xbar']
  y2 = df['R']
  ym = df['xbar'].mean()
  dm = df['R'].mean()
  ucl = ym+1.023*dm
  lcl = ym-1.023*dm
  UCL_R = 2.575*dm

  ax[0].plot(x,y,marker='o')
  ax[0].axhline(ucl, linestyle='--')
  ax[0].axhline(ym, linestyle='-')
  ax[0].axhline(lcl, linestyle='--')
  ax[0].set_ylim(40,80)
  ax[0].set_ylabel(r'$\bar{x}$')
  ax[0].text(51.5,ucl,"UCL:{0:.2f}".format(ucl))
  ax[0].text(51.5,ym,"CL:{0:.2f}".format(ym))
  ax[0].text(51.5,lcl,"LCL:{0:.2f}".format(lcl))
  ax[0].set_xticks(np.arange(0,55,5))


  ax[1].plot(x,y2, marker='x')
  ax[1].axhline(dm, linestyle='-')
  ax[1].axhline(UCL_R,linestyle='--')
  ax[1].set_ylim(0,20.0)
  ax[1].set_ylabel('R')
  ax[1].text(50.5,UCL_R,"UCL")
  ax[1].text(50.5,dm,"CL")

  plt.savefig("Graph4/Kanrizu_" + a + ".png", format="png", dpi=300)

  print(y.mean(), ucl, lcl, data_sigma*3+data_mean)
